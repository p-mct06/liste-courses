        <h3>👑 Administration</h3>

        <label for="newUser">Nouvel utilisateur autorisé :</label>

        <input type="email" id="newUser" placeholder="email@exemple.com">

        <button onclick="addUser()">Ajouter</button>

        <div id="userList"></div>

      </div>

    </div>

  </div>


  <script>

    const adminEmail = 'mouchet.family@gmail.com';

    let authorizedUsers = ['mouchet.family@gmail.com'];

    let currentUser = null;


    function login() {

      const email = document.getElementById('email').value.trim().toLowerCase();

      if (!authorizedUsers.includes(email)) {

        document.getElementById('loginError').textContent = "Accès refusé. Contactez l'administrateur.";

        return;

      }

      currentUser = email;

      localStorage.setItem('userEmail', email);

      showApp();

    }


    function showApp() {

      document.getElementById('loginSection').classList.add('hidden');

      document.getElementById('appSection').classList.remove('hidden');

      document.getElementById('userEmail').textContent = currentUser;

      if (currentUser === adminEmail) {

        document.getElementById('adminPanel').classList.remove('hidden');

        refreshUserList();

      }

    }


    function logout() {

      localStorage.removeItem('userEmail');

      location.reload();

    }


    function addUser() {

      const newEmail = document.getElementById('newUser').value.trim().toLowerCase();

      if (!newEmail || authorizedUsers.includes(newEmail)) return;

      authorizedUsers.push(newEmail);

      document.getElementById('newUser').value = '';

      refreshUserList();

    }


    function removeUser(email) {

      if (email === adminEmail) return;

      authorizedUsers = authorizedUsers.filter(user => user !== email);

      refreshUserList();

    }


    function refreshUserList() {

      const list = document.getElementById('userList');

      list.innerHTML = '';

      authorizedUsers.forEach(user => {

        const item = document.createElement('div');

        item.className = 'user-item';

        item.innerHTML = `<span>${user}</span>` +

          (user !== adminEmail ? `<button onclick="removeUser('${user}')">Supprimer</button>` : '<span style="color: #999; font-size: 12px;">(Admin)</span>');

        list.appendChild(item);

      });

    }


    // Auto-login if already connected

    window.onload = () => {

      const saved = localStorage.getItem('userEmail');

      if (saved && authorizedUsers.includes(saved)) {

        currentUser = saved;

        showApp();

      }

    };

  </script>
